package main

import (
    "fmt"
    "github.com/pinkai/sdk-go"
)

func main() {
    // Initialize AI Client
    client := pinkai.NewClient("YOUR_API_KEY")

    // Run analysis
    result, err := client.Analyze("Hello World Data")
    
    if err != nil {
        log.Fatal(err)
    }
    
    fmt.Printf("AI Score: %v\n", result.Score)
}
